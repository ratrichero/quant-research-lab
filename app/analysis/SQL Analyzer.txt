Kiểm tra winrate

SELECT 
    CASE 
        WHEN score >= 0 AND score < 3 THEN '0–3'
        WHEN score >= 3 AND score < 5 THEN '3–5'
        WHEN score >= 5 AND score < 7 THEN '5–7'
        WHEN score >= 7 AND score <= 10 THEN '7–10'
        ELSE 'Khác'
    END AS score_range,
    COUNT(*) AS total_signals,
    -- Tính tỉ lệ thắng trên tổng số lệnh đã đóng (WIN + LOSS)
    -- Nếu bạn muốn tính cả lệnh OPEN thì thay đổi công thức ở mẫu số
    ROUND(
        100.0 * COUNT(*) FILTER (WHERE status = 'WIN') / 
        NULLIF(COUNT(*) FILTER (WHERE status IN ('WIN', 'LOSS')), 0), 
        2
    ) AS winrate_percent
FROM signals
WHERE status IN ('WIN', 'LOSS') -- Chỉ xét các lệnh đã đóng để tính winrate chính xác
GROUP BY score_range
ORDER BY score_range;

1️⃣ COMPONENT-LEVEL PERFORMANCE
Phân tích từng thành phần xem có thực sự tạo edge không.

✅ 1.1 Winrate theo trend_score bucket
SQL

SELECT
    CASE
        WHEN sf.trend_score < 1 THEN 'LOW'
        WHEN sf.trend_score < 2 THEN 'MID'
        ELSE 'HIGH'
    END AS trend_bucket,

    COUNT(*) AS total_trades,

    ROUND(
        SUM(CASE WHEN s.status = 'WIN' THEN 1 ELSE 0 END) * 100.0
        / COUNT(*),
        2
    ) AS winrate_percent,

    ROUND(AVG(s.result_percent), 2) AS avg_return

FROM signal_features sf
JOIN signals s ON s.id = sf.signal_id

WHERE s.status IN ('WIN','LOSS')

GROUP BY trend_bucket
ORDER BY trend_bucket;

✅ 3.3 Profit Factor theo score bucket
SQL

SELECT
    CASE
        WHEN sf.total_score < 6 THEN 'LOW'
        ELSE 'HIGH'
    END AS bucket,

    SUM(CASE WHEN s.result_percent > 0 THEN s.result_percent ELSE 0 END) /
    NULLIF(ABS(SUM(CASE WHEN s.result_percent < 0 THEN s.result_percent ELSE 0 END)), 0)
    AS profit_factor

FROM signal_features sf
JOIN signals s ON s.id = sf.signal_id
WHERE s.status IN ('WIN','LOSS')
GROUP BY bucket;

Câu hỏi tiếp theo nên kiểm tra
Chạy thêm SQL này:

SQL

SELECT
    ROUND(AVG(total_score), 2),
    ROUND(AVG(CASE WHEN status='WIN' THEN total_score END), 2) AS avg_win_score,
    ROUND(AVG(CASE WHEN status='LOSS' THEN total_score END), 2) AS avg_loss_score
FROM signal_features sf
JOIN signals s ON s.id = sf.signal_id
WHERE s.status IN ('WIN','LOSS');
